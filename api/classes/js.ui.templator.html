Class Templator

Template class based on handlebars
模版工具类

Example:
let te = new Templator(), array = [];
for (let i = 0; i < 100000; i++) {
    array[i] = { val: i + 1 };
}
let text = te.compile('{{#.}}Number: {{val}}\n{{/.}}')(array);

});

});
